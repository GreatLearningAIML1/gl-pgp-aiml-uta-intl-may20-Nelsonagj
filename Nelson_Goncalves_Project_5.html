from scipy.spatial.distance import cdist
clusters=range(1,10)
meanDistortions=[]

for k in clusters:
    model=KMeans(n_clusters=k)
    model.fit(credit_df_z)
    prediction=model.predict(credit_df_z)
    meanDistortions.append(sum(np.min(cdist(credit_df_z, model.cluster_centers_, 'euclidean'), axis=1)) / credit_df_z
                           .shape[0])


plt.plot(clusters, meanDistortions, 'bx-')
plt.xlabel('k')
plt.ylabel('Average distortion')
plt.title('Selecting k with the Elbow Method')
